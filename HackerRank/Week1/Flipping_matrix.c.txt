#include<stdio.h>

int flippingMatrix(int matrix_rows, int matrix_columns, int** matrix) {
 int n=matrix_rows/2;
 int sum=0;
 for(int i=0;i<n;i++){
 for(int j=0;j<n;j++){
 int val1=matrix[i][j];
 int val2=matrix[i][2*n-1-j];
 int val3=matrix[2*n-1-i][j];
 int val4=matrix[2*n-1-i][2*n-1-j];
 int max_val=val1;
 if(val2>max_val) max_val=val2;
 if(val3>max_val) max_val=val3;
 if(val4>max_val) max_val=val4;

 sum+=max_val;

 }
 }
 return sum;

}
